### Знайти всі файли в поточній директорії, імена яких не прописані в list.txt.

#!/bin/bash

file="list.txt"

path=$(ls -F | grep -v /)

for var in $path

do

if !(grep -q "$var" $file)

then

echo "$var"

fi

done
